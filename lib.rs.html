67 /// letters are used (e.g. `F9B4CA`)

180 fn from_hex<T: AsRef<[u8]>>(hex: T) -> Result<Self, Self::Error> {
181 let hex = hex.as_ref();
182 if hex.len() % 2 != 0 {
183 return Err(FromHexError::OddLength);
184 }
185
186 hex.chunks(2)

233/// Encodes `data` as hex string using lowercase characters.
234///
235/// Lowercase characters are used (e.g. `f9b4ca`). The resulting string's
236/// length is always even, each byte in `data` is always encoded using two hex
237/// digits. Thus, the resulting string contains exactly twice as many bytes as
238/// the input data.
239///
240/// # Example
241///
242/// ```
243/// assert_eq!(hex::encode("Hello world!"), "48656c6c6f20776f726c6421");
244/// assert_eq!(hex::encode(vec![1, 2, 3, 15, 16]), "0102030f10");
245/// ```
246pub fn encode<T: AsRef<[u8]>>(data: T) -> String {
247 data.encode_hex()
248}

254/// # Example

258/// assert_eq!(hex::encode_upper(vec![1, 2, 3, 15, 16]), "0102030F10");

266/// Both, upper and lower case characters are valid in the input string and can

284/// Decode a hex string into a mutable bytes slice.
285///
286/// Both, upper and lower case characters are valid in the input string and can
287/// even be mixed (e.g. `f9b4ca`, `F9B4CA` and `f9B4Ca` are all valid strings).
288///
289/// # Example
290///
291/// ```
292/// let mut bytes = [0u8; 4];
293/// assert_eq!(hex::decode_to_slice("6b697769", &mut bytes as &mut [u8]), Ok(()));
294/// assert_eq!(&bytes, b"kiwi");
295/// ```
296pub fn decode_to_slice<T: AsRef<[u8]>>(data: T, out: &mut [u8]) -> Result<(), FromHexError> {
297 let data = data.as_ref();
298
299 if data.len() % 2 != 0 {
300 return Err(FromHexError::OddLength);
301 }
302 if data.len() / 2 != out.len() {
303 return Err(FromHexError::InvalidStringLength);
304 }
305
306 for (i, byte) in out.iter_mut().enumerate() {
307 *byte = val(data[2 * i], 2 * i)? << 4 | val(data[2 * i + 1], 2 * i + 1)?;
308 }
309
310 Ok(())
311}

331/// Encodes some bytes into a mutable slice of bytes.
332///
333/// The output buffer, has to be able to hold at least `input.len() * 2` bytes,
334/// otherwise this function will return an error.
335///
336/// # Example
337///
338/// ```
339/// # use hex::FromHexError;
340/// # fn main() -> Result<(), FromHexError> {
341/// let mut bytes = [0u8; 4 * 2];
342///
343/// hex::encode_to_slice(b"kiwi", &mut bytes)?;
344/// assert_eq!(&bytes, b"6b697769");
345/// # Ok(())
346/// # }
347/// ```
348pub fn encode_to_slice<T: AsRef<[u8]>>(input: T, output: &mut [u8]) -> Result<(), FromHexError> {
349 if input.as_ref().len() * 2 != output.len() {
350 return Err(FromHexError::InvalidStringLength);
351 }
352
353 for (byte, (i, j)) in input.as_ref().iter().zip(generate_iter(input.as_ref().len() * 2)) {
354 let (high, low) = byte2hex(*byte, HEX_CHARS_LOWER);
355 output[i] = high;
356 output[j] = low;
357 }
358
359 Ok(())
360}
